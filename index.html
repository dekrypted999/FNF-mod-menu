(function(){
  if(window.__fnfModMenuActive){alert("Menu already active!"); return;}
  window.__fnfModMenuActive=true;

  const div=document.createElement('div');
  div.style.position='fixed';
  div.style.top='50px';
  div.style.left='50px';
  div.style.width='220px';
  div.style.background='rgba(0,0,0,0.85)';
  div.style.color='#fff';
  div.style.border='2px solid #fff';
  div.style.padding='8px';
  div.style.borderRadius='10px';
  div.style.zIndex=9999;
  div.style.cursor='move';
  div.innerHTML = '<div style="text-align:center;font-weight:bold;margin-bottom:6px;">ðŸŽµ FNF Mod Menu</div>' +
                  '<button id="fnfMaxScore">ðŸŽ¯ Max Score</button>' +
                  '<button id="fnfInfiniteHealth">ðŸ’– Infinite Health</button>' +
                  '<button id="fnfFullCombo">ðŸ”¥ Full Combo</button>';
  document.body.appendChild(div);

  let drag=false, ox=0, oy=0;
  div.onmousedown=function(e){drag=true;ox=e.clientX-div.offsetLeft;oy=e.clientY-div.offsetTop;};
  document.onmousemove=function(e){if(drag){div.style.left=(e.clientX-ox)+'px';div.style.top=(e.clientY-oy)+'px';}};
  document.onmouseup=function(){drag=false;};

  let infiniteHealth=false;
  let maxScore=false;

  function getGameState(){
    for(let k in window){
      try{
        const v=window[k];
        if(v && typeof v==='object'){
          if(v.health!==undefined && v.score!==undefined) return v;
        }
      }catch(e){}
    }
    return null;
  }

  setInterval(()=>{
    const state=getGameState();
    if(!state) return;
    if(infiniteHealth && state.health!==undefined) state.health=2.0;
    if(maxScore && state.score!==undefined) state.score=Math.max(state.score||0,999999);
  },100);

  document.getElementById('fnfInfiniteHealth').onclick = ()=>{
    infiniteHealth = !infiniteHealth;
    alert('Infinite Health: '+(infiniteHealth?'ON':'OFF'));
  };
  document.getElementById('fnfMaxScore').onclick = ()=>{
    maxScore = !maxScore;
    alert('Max Score: '+(maxScore?'ON':'OFF'));
  };
  document.getElementById('fnfFullCombo').onclick = ()=>{
    const state=getGameState();
    if(state){
      if(state.combo!==undefined) state.combo=999;
      if(state.score!==undefined) state.score=999999;
      alert('Full Combo applied!');
    } else alert('Game state not found.');
  };
})();
